 39:  * @copyright microuser 2014, macro_user@outlook.com
 40:  */

 43:     /**
 44:      * The array of with the debuglog is stored
 45:      * @var array 
 46:      */
 47:     private $debugLog = array();
 48: 
 49:     public function __construct() {
 50:         $this->getConfig('DEBUG',$this->getConfig('envifronment','debugLevel'));
 51:     }

 53:     /**
 54:      * Appends text to the persisted global variable array named debugLog.  It also performs a backtrace such that the calling class and time are written to the log. For appropariate values of the bitmask, see the CONSTANTS such as:     DEBUG_ECHO,    DEBUG_ERROR_LOG,    DEBUG_VERBOSE,    DEBUG_FUNCTION_TRACE,    DEBUG_RETURN_VALUE,    DEBUG_DECISION.
 55:      * Also, if the bitmask of DEBUG has DEBUG_ERROR_LOG bit set, then the debug will be written with error_log() to the file set in php.ini
 56:      * @param int $bitmask
 57:      * @param String $text
 58:      */
 59:     public function addDebug($bitmask = null, $text = '') {
 60:         if ($bitmask === null) {
 61:             $bitmask = DEBUG;
 62:         }
 63:         if ($bitmask & DEBUG) {
 64:             $time = microtime(1);
 65:             $messageToWrite = self::getCallingFunctionName() . ": \t " . $text;
 66:             $this->debugLog[] = date("j/d/y\@H:i:s.", $time) . ($time * 1000) % 1000 . " \t " . $messageToWrite;
 67:             if (DEBUG & DEBUG_ERROR_LOG) {
 68:                 error_log($messageToWrite);
 69:             }
 70:         }
 71:     }

 73:     /**
 74:      * Perform a debug_backtrace() and infer the calling function and its class.
 75:      * @param DebugBacktrace $completeTrace
 76:      * @return String
 77:      */
 78:     private function getCallingFunctionName($completeTrace = false) {
 79:         $trace = debug_backtrace();
 80:         if ($completeTrace) {
 81:             $str = '';
 82:             foreach ($trace as $caller) {
 83:                 $str .= " -- {$caller['function']}";
 84:                 if (isset($caller['class']))
 85:                     $str .= "() of {$caller['class']}";
 86:             }
 87:         }
 88:         else {
 89:             if (!isset($trace[2])) {
 90:                 $str = ''; //No function

 92:                 $caller = $trace[2];
 93:                 $str = "{$caller['function']}";
 94:                 if (isset($caller['class']))
 95:                     $str .= "() of {$caller['class']}";
 96:             }
 97:         }
